Type Reference

TypeScript interfaces and types exported from calimero-client

Client Configuration

interface ClientConfig {
  nodeUrl: string;      // Calimero node base URL
  contextId: string;    // Active context ID
  timeout?: number;     // Request timeout ms (default 30000)
}

interface WsClientConfig extends ClientConfig {
  reconnect?: boolean;  // Auto-reconnect on disconnect (default true)
  reconnectMs?: number;// Delay between reconnects (default 2000)
}

RPC Types

interface RpcRequest {
  method: string;
  args: Record<string, unknown>;
}

interface RpcResponse<T = unknown> {
  result?: T;
  error?: {
    code: number;
    message: string;
  };
}

Event Types

interface NodeEvent {
  contextId: string;
  eventType: string;
  payload: unknown;
  timestamp: number;
}

interface SubscribeOptions {
  onEvent: (event: NodeEvent) => void;
  onError?: (err: Error) => void;
  onClose?: () => void;
}

type Unsubscribe = () => void;

Admin Data Types

interface Context {
  id: string;
  applicationId: string;
  rootHash: string;
}

interface Application {
  id: string;
  name: string;
  version: string;
  metadata: { links: { frontend?: string } };
}

interface Identity {
  publicKey: string;
}
