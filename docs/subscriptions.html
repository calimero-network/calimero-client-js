Subscriptions

Real-time event streaming with SubscriptionsClient over WebSocket

Setup

import { WsSubscriptionsClient } from '@calimero-network/calimero-client';

const subs = new WsSubscriptionsClient({
  nodeUrl: 'ws://localhost:2428',// WebSocket URL
  contextId: 'ctx_abc...',
});

subs.setToken(accessToken);

Subscribing to Events

// Subscribe — returns an unsubscribe function
const unsub = subs.subscribe({
  onEvent: (event) => {
    console.log('Received:', event.eventType, event.payload);
    if (event.eventType === 'StateMutation') {
      reloadData();
    }
  },
  onError: (err) => console.error('WS error:', err),
  onClose: () => console.log('Connection closed'),
});

// Stop and close connection:
unsub();

Event Shape

interface NodeEvent {
  contextId: string;
  eventType: string;  // 'StateMutation', 'ContextCreated', …
  payload: unknown; // event-specific data (JSON)
  timestamp: number;// Unix timestamp ms
}

React Integration Pattern

function useNodeEvents(contextId: string) {
  const [events, setEvents] = useState<NodeEvent[]>([]);

  useEffect(() => {
    const subs = new WsSubscriptionsClient({
      nodeUrl: 'ws://localhost:2428',
      contextId,
    });
    const unsub = subs.subscribe({
      onEvent: (e) => setEvents(prev => [...prev, e]),
    });
    return unsub; // cleanup on unmount
  }, [contextId]);

  return events;
}
